package com.dynamicg.common;

import android.util.Log;

/*
 * this class replaces the actual Logger for production builds
 * see build script
 */
public class Logger {

	public static final boolean isDebugEnabled = false;
	public static final boolean isTraceEnabled = false;
	
	public Logger(Class<?> cls) {}
	public void warn(String text, Object... args) {}
	public void debug(String text, Object... args) {}

	public static void dumpIfDevelopment(Throwable e) {}
	
}
